import java.io.IOException;
import java.io.PrintWriter;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Servlet implementation class ElectriCityBill
 */
@WebServlet("/ElectriCityBill")
public class ElectriCityBill extends HttpServlet {
	private static final long serialVersionUID = 1L;
    public ElectriCityBill() {
        super();
    }
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		response.setContentType("text/html");
		PrintWriter out=response.getWriter();
		int cid=Integer.parseInt(request.getParameter("cid"));
		String name=request.getParameter("cname");
		int cr=Integer.parseInt(request.getParameter("cr"));
		int pr=Integer.parseInt(request.getParameter("pr"));
		out.print("consumer id="+cid+"<br>");
		out.print("consumer name="+name+"<br>");
		out.print("current reading="+cr+"<br>");
		out.print("privious reading="+pr+"<br>");
		int units=cr-pr;
		double bill=0.0;
		out.print("units="+units+"<br>");
		if(units<300)
		{
			bill=units*1.75;
		}
		else if(units>=300 && units<600)
		{
			bill=units*5.75;
		}
		if(units>=600)
		{
			bill=units*7.25;
		}
		out.print("Total Bill="+bill);
	}

}
